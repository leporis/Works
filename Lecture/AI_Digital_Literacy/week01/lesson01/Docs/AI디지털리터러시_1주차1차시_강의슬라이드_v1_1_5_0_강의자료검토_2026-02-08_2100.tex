\section{\emoji{green-book} 전개 2: AI 유형}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{약한 AI vs 강한 AI}
\begin{columns}
\begin{column}{0.48\textwidth}
\begin{block}{\emoji{small-blue-diamond} 약한 AI (Narrow AI)}
\begin{itemize}
    \item 특정 작업 하나에 특화
    \item 현재 \textbf{모든 AI}가 약한 AI
    \item 시리, ChatGPT, 자율주행 등
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{\emoji{small-orange-diamond} 강한 AI (AGI)}
\begin{itemize}
    \item 인간처럼 범용적 사고
    \item \textbf{아직 실현되지 않음}
    \item 영화 속 AI (자비스 등)
\end{itemize}
\end{block}
\end{column}
\end{columns}

\vspace{0.5cm}
\begin{alertblock}{\emoji{light-bulb} 쉬운 비유}
약한 AI = ``한 과목만 잘하는 학생'' / 강한 AI = ``모든 과목 다 잘하는 학생''
\end{alertblock}
\end{frame}

\begin{frame}{ChatGPT도 약한 AI?}
\begin{center}
\Large ChatGPT는 대화도 하고, 글도 쓰고, 코딩도 하는데...\\

\pause
\textbf{\color{primary}텍스트 생성}이라는 \textbf{하나의 작업}에 특화된 약한 AI\\

\normalsize 그 하나의 작업이 워낙 범용적이라\\다재다능해 \textbf{보이는} 거예요
\end{center}
\end{frame}
