```javascript
const reward = chosen.lines ** 2 * 10      // 라인 클리어 (제곱)
             + (pf[1] - chosen.feat[1]) * 5  // 구멍 감소
             + (pf[0] - chosen.feat[0]) * 2  // 높이 감소
             + (pf[2] - chosen.feat[2]) * 1  // 울퉁불퉁함 감소
```
