\section{필터 사용법}
%===============================================================================

\subsection{필터 활성화}

\begin{enumerate}
\item \textbf{1행 선택} (헤더 행)
\item \textbf{데이터} 탭 클릭
\item \textbf{필터} 버튼 클릭
\item 각 열 헤더에 $\blacktriangledown$ 드롭다운 버튼 생성됨
\end{enumerate}

\subsection{특정 센서 포함 조합 찾기}

\textbf{예시: SHT40 (온습도 센서) 포함 조합 찾기}

\begin{enumerate}
\item D열(SHT40) 헤더의 $\blacktriangledown$ 클릭
\item ``전체 선택'' 해제
\item \textbf{●} 만 선택
\item 확인 클릭
\end{enumerate}

\begin{tcolorbox}[tipbox, title={결과}]
SHT40이 포함된 \textbf{127개 조합}만 표시됩니다.
\end{tcolorbox}

\subsection{복수 센서 조합 찾기}

\textbf{예시: SHT40 + BMP280 둘 다 포함된 조합 찾기}

\begin{enumerate}
\item D열(SHT40) 필터 → ● 선택
\item E열(BMP280) 필터 → ● 선택
\end{enumerate}

$\rightarrow$ \textbf{두 센서가 모두 포함된 조합만 표시}
